核心实现思路

// 1. 将文本拆分为字符，每个字符包裹在 <span class="char"> 中
// 2. 默认 .char 为 display: none（隐藏）
// 3. 通过 setTimeout 逐个添加 .is-typed 类（display: inline）
// 4. 添加光标元素，配合 CSS 动画实现闪烁

function typewriter(element, text, delay = 55) {
  element.textContent = '';
  
  // 拆分字符
  const chars = [];
  for (const ch of text) {
    const span = document.createElement('span');
    span.className = 'char';
    span.textContent = ch;
    element.appendChild(span);
    chars.push(span);
  }
  
  // 添加光标
  const caret = document.createElement('span');
  caret.className = 'type-caret';
  element.appendChild(caret);
  
  // 逐个显示
  chars.forEach((c, i) => {
    setTimeout(() => c.classList.add('is-typed'), i * delay);
  });
  
  // 完成后隐藏光标
  setTimeout(() => element.classList.add('is-done'), chars.length * delay + 150);
}
